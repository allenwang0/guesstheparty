<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>119th Congress - Blind Guess</title>

    <script src="https://cdn.tailwindcss.com"></script>

    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <script src="https://unpkg.com/lucide@latest"></script>

</head>

<body class="bg-[#050505] text-white font-sans antialiased overflow-hidden">

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useMemo } = React;


        const POLITICIANS = [

            { name: "Yadira Caraveo", pos: "U.S. Rep", state: "Colorado", party: "Democrat", id: "C001117" },

            { name: "Lori Chavez-DeRemer", pos: "U.S. Rep", state: "Oregon", party: "Republican", id: "C001129" },

            { name: "Don Davis", pos: "U.S. Rep", state: "NC", party: "Democrat", id: "D000230" },

            { name: "Juan Ciscomani", pos: "U.S. Rep", state: "Arizona", party: "Republican", id: "C001133" },

            { name: "Marie Perez", pos: "U.S. Rep", state: "Washington", party: "Democrat", id: "G000600" },

            { name: "Jen Kiggans", pos: "U.S. Rep", state: "Virginia", party: "Republican", id: "K000399" },

            { name: "Mary Peltola", pos: "U.S. Rep", state: "Alaska", party: "Democrat", id: "P000619" },

            { name: "Marc Molinaro", pos: "U.S. Rep", state: "New York", party: "Republican", id: "M001221" },

            { name: "Gabe Vasquez", pos: "U.S. Rep", state: "New Mexico", party: "Democrat", id: "V000135" },

            { name: "Anna Paulina Luna", pos: "U.S. Rep", state: "Florida", party: "Republican", id: "L000596" },

            { name: "Brittany Pettersen", pos: "U.S. Rep", state: "Colorado", party: "Democrat", id: "P000620" },

            { name: "Mike Lawler", pos: "U.S. Rep", state: "New York", party: "Republican", id: "L000599" },

            { name: "Hillary Scholten", pos: "U.S. Rep", state: "Michigan", party: "Democrat", id: "S001221" },

            { name: "Zach Nunn", pos: "U.S. Rep", state: "Iowa", party: "Republican", id: "N000193" },

            { name: "Val Hoyle", pos: "U.S. Rep", state: "Oregon", party: "Democrat", id: "H001094" },

            { name: "John Duarte", pos: "U.S. Rep", state: "California", party: "Republican", id: "D000633" }

        ];


        const App = () => {

            const [deck, setDeck] = useState([]);

            const [idx, setIdx] = useState(0);

            const [score, setScore] = useState(0);

            const [revealed, setRevealed] = useState(false);

            const [isCorrect, setIsCorrect] = useState(null);

            const [gameOver, setGameOver] = useState(false);

            const [isTransitioning, setIsTransitioning] = useState(false);


            useEffect(() => {

                setDeck([...POLITICIANS].sort(() => Math.random() - 0.5));

            }, []);


            const handleGuess = (guess) => {

                if (revealed || isTransitioning) return;

                const correct = deck[idx].party === guess;

                setIsCorrect(correct);

                if (correct) setScore(s => s + 1);

                setRevealed(true);

            };


            const nextMember = () => {

                // 1. Start transition (fades out the whole card immediately)

                setIsTransitioning(true);

                

                // 2. Short timeout to allow fade-out before data swap

                setTimeout(() => {

                    if (idx + 1 < deck.length) {

                        setRevealed(false);

                        setIsCorrect(null);

                        setIdx(prev => prev + 1);

                        setIsTransitioning(false);

                    } else {

                        setGameOver(true);

                    }

                }, 300); 

            };


            if (deck.length === 0) return null;

            const current = deck[idx];

            const imgUrl = `https://www.congress.gov/img/member/${current.id.toLowerCase()}_200.jpg`;


            return (

                <div className="min-h-screen flex items-center justify-center p-4">

                    <div className="w-full max-w-sm flex flex-col gap-6">

                        

                        {/* Header Stats */}

                        <div className="flex justify-between items-center px-1">

                            <div className="space-y-1">

                                <h1 className="text-[10px] font-black tracking-[0.3em] text-blue-500 uppercase">119th Congress</h1>

                                <div className="flex gap-1">

                                    {deck.map((_, i) => (

                                        <div key={i} className={`h-1 w-2 rounded-full transition-all duration-300 ${i === idx ? 'bg-blue-500 w-6' : i < idx ? 'bg-slate-700' : 'bg-slate-900'}`} />

                                    ))}

                                </div>

                            </div>

                            <div className="text-right">

                                <span className="text-2xl font-mono font-bold">{score}</span>

                                <span className="text-slate-600 font-bold ml-1">/{idx + (revealed ? 1 : 0)}</span>

                            </div>

                        </div>


                        {/* Card Container with unique key to force reset */}

                        <div 

                            key={idx} 

                            className={`relative aspect-[3/4] rounded-[48px] overflow-hidden bg-slate-900 border border-white/10 shadow-2xl transition-opacity duration-300 ${isTransitioning ? 'opacity-0' : 'opacity-100'}`}

                        >

                            <img 

                                src={imgUrl} 

                                className={`w-full h-full object-cover transition-all duration-500 ${revealed ? 'scale-110 blur-xl brightness-50' : 'scale-100'}`} 

                                alt="Politician" 

                            />

                            

                            {/* Hidden Details (Reveals strictly on Guess) */}

                            <div className={`absolute inset-0 flex flex-col justify-end p-10 transition-all duration-500 ${revealed ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-10 pointer-events-none'}`}>

                                <div className="space-y-3">

                                    <span className={`inline-block px-4 py-1 rounded-full text-[10px] font-black uppercase tracking-widest ${current.party === 'Democrat' ? 'bg-blue-600' : 'bg-red-600'}`}>

                                        {current.party}

                                    </span>

                                    <h2 className="text-4xl font-bold leading-tight">{current.name}</h2>

                                    <div className="flex items-center gap-4 text-slate-400 text-xs font-semibold">

                                        <span>{current.pos}</span>

                                        <span>•</span>

                                        <span>{current.state}</span>

                                    </div>

                                    <button 

                                        onClick={nextMember}

                                        className="mt-6 w-full py-4 bg-white text-black rounded-3xl font-bold flex items-center justify-center gap-2 hover:bg-slate-200 active:scale-95 transition-all"

                                    >

                                        Next Member

                                    </button>

                                </div>

                            </div>


                            {/* Feedback Indicators */}

                            {revealed && (

                                <div className="absolute top-10 right-10">

                                    {isCorrect ? 

                                        <span className="text-green-500 text-6xl drop-shadow-lg">✓</span> : 

                                        <span className="text-red-500 text-6xl drop-shadow-lg">✕</span>

                                    }

                                </div>

                            )}

                        </div>


                        {/* Interaction Buttons */}

                        {!revealed ? (

                            <div className="grid grid-cols-2 gap-4">

                                <button onClick={() => handleGuess('Democrat')} className="py-6 rounded-[32px] bg-blue-600/10 border border-blue-500/30 text-blue-400 font-black text-xs tracking-widest hover:bg-blue-600 hover:text-white transition-all active:scale-90 uppercase">

                                    Democrat

                                </button>

                                <button onClick={() => handleGuess('Republican')} className="py-6 rounded-[32px] bg-red-600/10 border border-red-500/30 text-red-400 font-black text-xs tracking-widest hover:bg-red-600 hover:text-white transition-all active:scale-90 uppercase">

                                    Republican

                                </button>

                            </div>

                        ) : (

                            <div className="h-[76px]" />

                        )}

                    </div>


                    {/* Result Screen */}

                    {gameOver && (

                        <div className="fixed inset-0 bg-black/95 backdrop-blur-xl z-50 flex flex-col items-center justify-center p-8 text-center">

                            <h2 className="text-5xl font-black mb-2 uppercase">Score</h2>

                            <p className="text-8xl font-black text-blue-500 mb-10">{Math.round((score/deck.length)*100)}%</p>

                            <button onClick={() => window.location.reload()} className="px-12 py-5 bg-white text-black rounded-full font-black tracking-widest hover:bg-slate-200 active:scale-95 transition-all">

                                PLAY AGAIN

                            </button>

                        </div>

                    )}

                </div>

            );

        };


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>

</body>
